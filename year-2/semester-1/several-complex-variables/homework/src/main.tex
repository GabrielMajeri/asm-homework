% Swap the definition of \ceil*, \abs* and \norm*, so that \abs

% starred version does not.

\let\oldceil\ceil
\def\ceil{\@ifstar{\oldceil}{\oldceil*}}

\def\abs{\@ifstar{\oldabs}{\oldabs*}}
%
\let\oldnorm\norm
\def\norm{\@ifstar{\oldnorm}{\oldnorm*}}
\makeatother
